var animals = ['cats', 'dogs'];
function makeEmRock(animals) {
  return animals.map(function(animal) {
    return animal + ' rock';
  });
}
var rockinAnimals = makeEmRock(animals);
// rockinAnimals = [cats rock, dogs rock]
// animals = [cats, dogs]
