iD.util.asyncMap = function(inputs, func, callback) {
  var remaining = inputs.length,
      results = [], errors = [];
  inputs.forEach(function(d, i) {
    func(d, function done(err, data) {
      errors[i] = err;
      results[i] = data;
      remaining --;
      if (!remaining) callback(errors, results);
    });
  });
};
